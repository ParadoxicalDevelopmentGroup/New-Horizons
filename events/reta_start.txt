namespace = reta_start

event = {
	id = reta_start.1
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		every_country = {
			limit = { exists = capital_scope }
			
			capital_scope = {
				if = {
					limit = {
						and = {
							is_planet_class = "pc_gaia"
							not = {has_trait = "trait_pc_desert_preference"}
							not = {has_trait = "trait_pc_tropical_preference"}
							not = {has_trait = "trait_pc_arid_preference"}
							not = {has_trait = "trait_pc_continental_preference"}
							not = {has_trait = "trait_pc_ocean_preference"}
							not = {has_trait = "trait_pc_tundra_preference"}
							not = {has_trait = "trait_pc_arctic_preference"}
						}
					}
					random_list = {
						14 = { 
							change_pc = "pc_desert" 
							modify_species = {
								species = OWNER
								add_trait = "trait_pc_desert_preference"
								remove_trait = "trait_pc_gaia_preference"
							}
						}
						14 = { 
							change_pc = "pc_tropical" 
							modify_species = {
								species = OWNER
								add_trait = "trait_pc_tropical_preference"
								remove_trait = "trait_pc_gaia_preference"
							}
						}
						14 = { 
							change_pc = "pc_arid" 
							modify_species = {
								species = OWNER
								add_trait = "trait_pc_arid_preference"
								remove_trait = "trait_pc_gaia_preference"
							}
						}
						14 = { 
							change_pc = "pc_continental" 
							modify_species = {
								species = OWNER
								add_trait = "trait_pc_continental_preference"
								remove_trait = "trait_pc_gaia_preference"
							}
						}
						14 = { 
							change_pc = "pc_ocean" 
							modify_species = {
								species = OWNER
								add_trait = "trait_pc_ocean_preference"
								remove_trait = "trait_pc_gaia_preference"
							}
						}
						14 = { 
							change_pc = "pc_tundra" 
							modify_species = {
								species = OWNER
								add_trait = "trait_pc_tundra_preference"
								remove_trait = "trait_pc_gaia_preference"
							}
						}
						14 = { 
							change_pc = "pc_arctic" 
							modify_species = {
								species = OWNER
								add_trait = "trait_pc_arctic_preference"
								remove_trait = "trait_pc_gaia_preference"
							}
						}
					}
				}
			}
			
			if = {
				limit = {ideal_planet_class = "pc_desert"}
				capital_scope = {
					change_pc = "pc_desert"
				}
			}
			if = {
				limit = {ideal_planet_class = "pc_tropical"}
				capital_scope = {
					change_pc = "pc_tropical"
				}
			}
			if = {
				limit = {ideal_planet_class = "pc_arid"}
				capital_scope = {
					change_pc = "pc_arid"
				}
			}
			if = {
				limit = {ideal_planet_class = "pc_continental"}
				capital_scope = {
					change_pc = "pc_continental"
				}
			}
			if = {
				limit = {ideal_planet_class = "pc_ocean"}
				capital_scope = {
					change_pc = "pc_ocean"
				}
			}
			if = {
				limit = {ideal_planet_class = "pc_tundra"}
				capital_scope = {
					change_pc = "pc_tundra"
				}
			}
			if = {
				limit = {ideal_planet_class = "pc_arctic"}
				capital_scope = {
					change_pc = "pc_arctic"
				}
			}
		}
	}
}
